82void StateMachine::GarbageCollectorCallback(void* parametersP) {
83 bool exit = false;
84
85#ifdef SM_TRACES
86 LogText(SM_MODULE, 0, true, "State Machine garbage collector is starting");
87#endif
88
89 while (!exit) {
90 sleep(GARBAGE_COLLECTOR_PAUSE);
91
92 // go through all the state machines and delete those that are declared as exiting
93 list<StateMachine*>::iterator i = m_machines.begin();
94 while (i != m_machines.end()) {
95 StateMachine* smP = *i++;
96
97 if (smP->IsExiting() && !smP->IsPendingThread()) {
98#ifdef SM_TRACES
99 LogVText(SM_MODULE, 4, true, "State Machine garbage collector is deleting sm %p", smP);
100#endif
101 m_machines.remove(smP);
102 delete smP;
103 }
104 }
105
106 if (m_machines.empty())
107 exit = true;
108 }
109
110#ifdef SM_TRACES
111 LogText(SM_MODULE, 0, true, "State Machine garbage collector is exiting");
112#endif
113
114 return;
115}

124 CallbackThreadParameters* paramsP = (CallbackThreadParameters*)parametersP;
125 if (!paramsP || !paramsP->m_machineP || paramsP->m_machineP->IsExiting()) {

163void StateMachine::TimerThreadCallback(void* parametersP) {
164 TimerThreadParameters* paramsP = (TimerThreadParameters*)parametersP;
165
166 if (!paramsP || !paramsP->m_machineP)
167 return;
168
169#ifdef SM_TRACES
170 LogVText(SM_MODULE, 0, true, "timerThread sleeping for %ld milliseconds", paramsP->m_milli_seconds);
171#endif
172
173 // if 0 returned, the cancelled semaphore was inc'ed by the state machine
174 if (paramsP->m_cancelled.WaitA(paramsP->m_milli_seconds)) {
175
176 paramsP->m_machineP->m_timer_sem.A();
177
178 if (!paramsP->m_machineP->m_timer_thread.IsAskedToStop()) {
179 #ifdef SM_TRACES
180 LogText(SM_MODULE, 0, true, "timeout!!!");
181 #endif
182 paramsP->m_machineP->m_timer_sem.R();
183 paramsP->m_machineP->DoTransition(paramsP->m_transition_name, true);
184 } else
185 paramsP->m_machineP->m_timer_sem.R();
186 }
187#ifdef SM_TRACES
188 else LogText(SM_MODULE, 0, true, "TimerThread cancelled !!!");
189#endif
190
191 return;
192}

268 return NULL;

270

272 if (!sm.isMember("initial_state") || !sm["initial_state"].isString()) {
273 cerr << __FILE__ << ", " << __FUNCTION__ << "(" << __LINE__ << ") Error: StateMachine definition missig 'initial_state' object!" << endl;
274 return NULL;

279 return NULL;

305 return NULL;

399 }
400 }
401
402 // if a timer, set it
403 State* initialStateP = (State *)smP->GetState();
404 unsigned long timeOut = initialStateP->GetTimeout();
405 if (timeOut != 0)
406 smP->InitiateTransitionTimer(timeOut, initialStateP->GetTimeoutTransition());
407
408 } catch (Json::RuntimeError& e) {
409 cerr << __FILE__ << ", " << __FUNCTION__ << "(" << __LINE__ << ") Error:" << endl << e.what() << endl;
410 return NULL;
411 }
412
413 return smP;
