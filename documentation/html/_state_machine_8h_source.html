431 void DoTransition(const string &transition_name, bool from_timer_thread = false) {
432 if (!m_current_stateP || IsExiting())
433 return;
434
435 // cancel pending timer if any
436 if (!from_timer_thread)
437 CancelTransitionTimer();
438
439 // do inject the transition
440 string callback_name;
441 State *srcP;
442 if ((srcP = m_current_stateP))
443 m_current_stateP = m_current_stateP->DoTransition(transition_name, callback_name);
444
445 // if a timer, set it
446 if (m_current_stateP->GetTimeout() != 0)
447 InitiateTransitionTimer(m_current_stateP->GetTimeout(), m_current_stateP->GetTimeoutTransition());
448
449 // call into the callback if present
450 // the state machine may be dying when the callback thread calls back into it
451 // so, we need to keep a list of the threads, and wait for them when the SM dies
452 // before actually destroying the SM (see garbage collector)
453 Thread *threadP = new Thread(CallbackThread);
454
455 // keep the ref of this thread in the SM's thread list
456 m_call_sem.A();
457 m_callback_threads.push_back(threadP);
458 m_call_sem.R();
459
460 // start the callback thread
461 threadP->Run(new CallbackThreadParameters(threadP, this, callback_name, srcP, m_current_stateP, transition_name, m_user_dataP));
462 }
