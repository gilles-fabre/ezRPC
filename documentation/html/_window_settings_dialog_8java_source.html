365 private void addModules(int indices[]) {
366 int i;
367 String addNames[] = new String[indices.length];
368 for (i=0; i<indices.length; i++)
369 addNames[i] = globalModIds[indices[i]];
370 contentFrame.addModIdNames(addNames);
371 windowModIds = new String[contentFrame.getModIdNames().size()];
372 i=0;
373 for (String filterName : contentFrame.getModIdNames())
374 windowModIds[i++] = filterName;
375 }
376
377 private void delModules(int indices[]) {
378 int i,j,k;
379 String tmpModIds[] = new String[windowModIds.length - indices.length];
380 i=j=k=0;
381 while (k<indices.length) {
382 while (j<indices[k])
383 tmpModIds[i++] = windowModIds[j++];
384 j++;
385 k++;
386 }
387 while (i<tmpModIds.length)
388 tmpModIds[i++] = windowModIds[j++];
389 windowModIds = tmpModIds;
390 contentFrame.setModIdNames(windowModIds);
391 }
