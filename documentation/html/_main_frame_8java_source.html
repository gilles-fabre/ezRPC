444 private void showMemUsageDialog() {

698 private void editIndexedModId(int iModel, Component where) {
699 if (iModel >= 0) {
700 ModuleFilter mf = ModIdDefinitionDialog.doIt(this, where, filter.getModuleIdFilter(iModel));
701 if (mf != null) {
702 String oldName = (String)modIdTable.getModel().getValueAt(iModel, ModIdTableModel._NameColumn);
703 String newName = mf.getName();
704 System.out.format("new filter def = %s 0x%08x 0x%08x 0x%08x\n", newName, mf.getModMask(), mf.getModId(), mf.getColor().getRGB());
705 if (!oldName.equals(newName)) {
706 ContentFrameEnsemble.getInstance().changeModIdName(oldName, newName);
707 }
708 filter.setModuleIdFilter(iModel, mf);
709 setFilterAndUpdateContentFrameFilters(filter);
710 }

718 int rowCount = model.getRowCount();
719 for (int iModel = 0; iModel < rowCount; iModel++) {
