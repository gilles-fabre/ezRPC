272static void BuildMessage(Message* messageP, const char* appName, uint32_t module, uint8_t indent, const uint8_t* dataP, uint16_t size, TYPE_DATA_TYPE type) {
273 // get pid and tid.
274 uint32_t pid, tid;
275#ifdef WIN32
276 pid = (uint32_t)_getpid();
277 tid = (uint32_t)GetCurrentThreadId();
278
279#else
280 pid = (uint32_t)getpid();
281 tid = (uint32_t)syscall(SYS_gettid);
282
283#endif
284 // build message structure
285 // app name parameter
286 Parameter* appNameParamP = &messageP->parameters[APPNAME_PARAM];
287 appNameParamP->header.size = (uint16_t)strlen(appName) + 1;
288 appNameParamP->dataP = (uint8_t*)malloc(appNameParamP->header.size);
289 assert(appNameParamP->dataP != NULL);
290 memcpy(appNameParamP->dataP, appName, appNameParamP->header.size);
291 appNameParamP->header.type = APPNAME;
292
293 // attributes
294 Parameter* attrParamP = &messageP->parameters[ATTR_PARAM];
295 attrParamP->header.size = ATTR_DATA_SIZE;
296 attrParamP->dataP = (uint8_t*)malloc(ATTR_DATA_SIZE);
297 assert(attrParamP->dataP != NULL);
298 *(MODULE_DATA_TYPE*)(attrParamP->dataP) = (MODULE_DATA_TYPE)htonl(module);
299 *(TYPE_DATA_TYPE*)(&attrParamP->dataP[sizeof(MODULE_DATA_TYPE)]) = type;
300 attrParamP->header.type = ATTR;
301
302 // indent
303 Parameter* indentParamP = &messageP->parameters[INDENT_PARAM];
304 indentParamP->header.size = INDENT_DATA_SIZE;
305 indentParamP->dataP = (uint8_t*)malloc(INDENT_DATA_SIZE);
306 assert(indentParamP->dataP != NULL);
307 *(INDENT_DATA_TYPE*)(indentParamP->dataP) = indent;
308 indentParamP->header.type = INDENT;
309
310 // pid
311 Parameter* pidParamP = &messageP->parameters[PID_PARAM];
312 pidParamP->header.size = PID_DATA_SIZE;
313 pidParamP->dataP = (uint8_t*)malloc(PID_DATA_SIZE);
314 assert(pidParamP->dataP != NULL);
315 *(PID_DATA_TYPE*)(pidParamP->dataP) = (PID_DATA_TYPE)htonl(pid);
316 pidParamP->header.type = PID;
317
318 // tid
319 Parameter* tidParamP = &messageP->parameters[TID_PARAM];
320 tidParamP->header.size = TID_DATA_SIZE;
321 tidParamP->dataP = (uint8_t*)malloc(TID_DATA_SIZE);
322 assert(tidParamP->dataP != NULL);
323 *(TID_DATA_TYPE*)(tidParamP->dataP) = (TID_DATA_TYPE)htonl(tid);
324 tidParamP->header.type = TID;
325
326 // data
327 Parameter* dataParamP = &messageP->parameters[DATA_PARAM];
328 dataParamP->header.size = size;
329 dataParamP->dataP = (uint8_t*)malloc(dataParamP->header.size);
330 assert(dataParamP->dataP != NULL);
331 memcpy(dataParamP->dataP, dataP, dataParamP->header.size);
332 dataParamP->header.type = DATA;
333
334 // timestamp
335 Parameter* timeParamP = &messageP->parameters[TIME_PARAM];
336 timeParamP->header.size = TIME_DATA_SIZE;
337 timeParamP->dataP = (uint8_t*)malloc(TIME_DATA_SIZE);
338 assert(timeParamP->dataP != NULL);
339 *(TIME_DATA_TYPE*)(timeParamP->dataP) =
340 (TIME_DATA_TYPE)htonll((TIME_DATA_TYPE)GetMilliTime());
341 timeParamP->header.type = TIME;
342
343 // eom
344 Parameter* eomParamP = &messageP->parameters[EOM_PARAM];
345 eomParamP->header.size = 0;
346 eomParamP->dataP = NULL;
347 eomParamP->header.type = EOM;
348
349 // create message
350 messageP->header.version = (uint16_t)((uint16_t)PROTOCOL_VERSION_MAJOR << 8 |
351 (uint16_t)PROTOCOL_VERSION_MINOR);
352 messageP->header.size = MESSAGE_HEADER_SIZE +
353 appNameParamP->header.size + PARAM_HEADER_SIZE +
354 attrParamP->header.size + PARAM_HEADER_SIZE +
355 indentParamP->header.size + PARAM_HEADER_SIZE +
356 pidParamP->header.size + PARAM_HEADER_SIZE +
357 tidParamP->header.size + PARAM_HEADER_SIZE +
358 dataParamP->header.size + PARAM_HEADER_SIZE +
359 timeParamP->header.size + PARAM_HEADER_SIZE +
360 PARAM_HEADER_SIZE;
361}

371static void SendAndFreeMessage(Message* messageP, uint8_t logToFile) {
372 uint32_t u32;
373 uint16_t u16;
374
375 if (!messageP)
376 return;
377
378 // send complete message because the peer might not parse properly incomplete messages..
379 uint8_t* bufferP = (uint8_t*)malloc(messageP->header.size);
380 assert(bufferP != NULL);
381 int i = 0;
382
383 // protocol version
384 u16 = htons(messageP->header.version);
385 memcpy(&bufferP[i], &u16, sizeof(u16));
386 i += sizeof(u16);
387
388 // message size
389 u32 = htonl(messageP->header.size);
390 memcpy(&bufferP[i], &u32, sizeof(u32));
391 i += sizeof(u32);
392
393 // parameters
394 int p = 0;
395 do {
396 bufferP[i++] = messageP->parameters[p].header.type;
397 u16 = htons(messageP->parameters[p].header.size);
398 memcpy(&bufferP[i], &u16, sizeof(u16));
399 i += sizeof(u16);
400 if (messageP->parameters[p].header.type != EOM) {
401 memcpy(&bufferP[i], messageP->parameters[p].dataP, messageP->parameters[p].header.size);
402 i += messageP->parameters[p].header.size;
403
404 // free allocated data buffer
405 free(messageP->parameters[p].dataP);
406 }
407 } while (messageP->parameters[p++].header.type != EOM);
408
409 // to tcp
410 if (gLogSocket != -1) {
411#ifdef WIN32
412 if (send((SOCKET)gLogSocket, (const char*)bufferP, messageP->header.size, 0) < 0) {
413#else
414 if (write(gLogSocket, bufferP, messageP->header.size) < 0) {
415#endif
416 gConnectionFailed = true;
417 gLogSocket = -1;
418 }
419 }
420
421 // to file
422 if (logToFile && gLogFile != -1) {
423 uint64_t pos = 0;
424
425 // check log size
426 if (gMaxLogSize != -1)
427 // stop logging to file if it went too big
428#ifdef WIN32
429 pos = _lseek(gLogFile, 0, SEEK_END);
430
431#else
432 pos = (uint64_t)lseek(gLogFile, 0, SEEK_END);
433
434#endif
435 if (gMaxLogSize == -1 || (pos >> 10) < gMaxLogSize) {
436#ifdef WIN32
437 _write(gLogFile, bufferP, messageP->header.size);
438#else
439 write(gLogFile, bufferP, messageP->header.size);
440#endif
441 }
442 }
443
444 // free allocated data buffer
445 free(bufferP);
446}

454static void DumpLogFile(const char* appName) {
455 if (gLogFile != -1 && gLogSocket != -1) {
456
457 // connected, if we were logging to file, re-send file content over new connection
458
459 // set file r/w offset to 0
460#ifdef WIN32
461 _lseek(gLogFile, 0, SEEK_SET);
462
463#else
464 lseek(gLogFile, 0, SEEK_SET);
465
466#endif
467 // send a 'start dump message, so reporter can filter out redundancies'
468 Message message;
469 BuildMessage(&message, appName, 0, 0, (uint8_t*)START_DUMP_MSG, (uint16_t)strlen(START_DUMP_MSG), START_DUMP_TAG);
470 SendAndFreeMessage(&message, 0);
471
472 // dump to reporter
473 int bytes;
474 unsigned char buffer[1024];
475 do {
476#ifdef WIN32
477 if ((bytes = _read(gLogFile, buffer, sizeof(buffer))) >= 1)
478 send(gLogSocket, (const char*)buffer, bytes, 0);
479#else
480 if ((bytes = read(gLogFile, buffer, sizeof(buffer))) >= 1)
481 write(gLogSocket, buffer, bytes);
482#endif
483 } while (bytes > 0);
484
485 // send a 'end dump message, so reporter can filter out redundancies'
486 BuildMessage(&message, appName, 0, 0, (uint8_t*)END_DUMP_MSG, (uint16_t)strlen(END_DUMP_MSG), END_DUMP_TAG);
487 SendAndFreeMessage(&message, 0);
488
489 // truncate file
490#ifdef WIN32
491 _chsize(gLogFile, 0);
492#else
493 ftruncate(gLogFile, 0);
494
495#endif
496 }
497}

537static int InitConnection(const char* appName) {
538 // if logging, keep going
539 if (gLogSocket != -1 || (!gAutoReconnect && gConnectionFailed))
540 return 0;
541
542#ifdef WIN32
543 if (!gWSAStartupDone) {
544 gWSAStartupDone = true;
545
546 // more windows crap..
547 #pragma comment( lib, "ws2_32.lib")
548 WSADATA wsa;
549 if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0)
550 return -1;
551 }
552#endif
553
554 struct sockaddr_in serverAddr;
555 socklen_t addr_size;
556
557 /*---- Create the socket. The three arguments are: ----*/
558 /* 1) Internet domain 2) Stream socket 3) Default protocol (TCP in this case) */
559#ifdef WIN32
560 SOCKET fd = socket(PF_INET, SOCK_STREAM, 0);
561
562#else
563 int fd = socket(PF_INET, SOCK_STREAM, 0);
564
565#endif
566 /*---- Configure settings of the server address struct ----*/
567 /* Address family = Internet */
568 serverAddr.sin_family = AF_INET;
569
570 /* Set port number, using htons function to use proper byte order */
571#ifdef WIN32
572 serverAddr.sin_port = (uint16_t)htons(gServerPort);
573
574#else
575 serverAddr.sin_port = htons(gServerPort);
576
577#endif
578 /* Set IP address to localhost */
579 serverAddr.sin_addr.s_addr = inet_addr(gServerIpAddress);
580
581 /* Set all bits of the padding field to 0 */
582 memset(serverAddr.sin_zero, '\0', sizeof serverAddr.sin_zero);
583
584 /*---- Connect the socket to the server using the address struct ----*/
585 addr_size = sizeof serverAddr;
586 if (!connect(fd, (struct sockaddr*)&serverAddr, addr_size)) {
587 gConnectionFailed = false;
588 gLogSocket = (int)fd;
589 DumpLogFile(appName);
590 } else
591 gConnectionFailed = true;
592
593 return gLogSocket > 0 ? 0 : -1;
594}
