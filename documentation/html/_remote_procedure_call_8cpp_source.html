51unsigned char* RemoteProcedureCall::ReceivePacket(unsigned long& data_len) {
52 uint64_t len;
53 unsigned char buffer[sizeof(uint64_t)];
54 unsigned char* bufferP;
55
56 // retrieve the packet len
57 m_linkP->Receive(buffer, (unsigned long)sizeof(len));
58
59 int offset = 0;
60 len = decode_uint64(buffer, offset);
61 if ((data_len = (unsigned long)NTOHLL(len)) == 0)
62 return NULL;
63
64 // allocate the necessary bufferP
65 if (!(bufferP = (unsigned char*)malloc(data_len)))
66 return NULL;
67
68 // wait and read the whole call stream
69 if (!m_linkP->Receive(bufferP, (unsigned long)data_len)) {
70 free(bufferP); // garbage received before connection dropped
71 bufferP = NULL;
72 }
73
74 return bufferP;
