354template <typename T>
355struct adl_serializer<std::shared_ptr<T>>
356{
357 static void to_json(json& j, const std::shared_ptr<T>& opt)
358 {
359 if (opt)
360 {
361 j = *opt;
362 }
363 else
364 {
365 j = nullptr;
366 }
367 }
368
369 static void from_json(const json& j, std::shared_ptr<T>& opt)
370 {
371 if (j.is_null())
372 {
373 opt = nullptr;
374 }
375 else
376 {
377 opt.reset(new T(j.get<T>())); // NOLINT(cppcoreguidelines-owning-memory)
378 }
379 }
380};

543// custom serializer - advanced usage
544// pack structs that are pod-types (but not scalar types)
545// relies on adl for any other type
546template <typename T, typename = void>
547struct pod_serializer
548{
549 // use adl for non-pods, or scalar types
550 template <
551 typename BasicJsonType, typename U = T,
552 typename std::enable_if <
553 !(std::is_pod<U>::value && std::is_class<U>::value), int >::type = 0 >
554 static void from_json(const BasicJsonType& j, U& t)
555 {
556 using nlohmann::from_json;
557 from_json(j, t);
558 }
559
560 // special behaviour for pods
561 template < typename BasicJsonType, typename U = T,
562 typename std::enable_if <
563 std::is_pod<U>::value && std::is_class<U>::value, int >::type = 0 >
564 static void from_json(const BasicJsonType& j, U& t)
565 {
566 std::uint64_t value = 0;
567 // The following block is no longer relevant in this serializer, make another one that shows the issue
568 // the problem arises only when one from_json method is defined without any constraint
569 //
570 // Why cannot we simply use: j.get<std::uint64_t>() ?
571 // Well, with the current experiment, the get method looks for a from_json
572 // function, which we are currently defining!
573 // This would end up in a stack overflow. Calling nlohmann::from_json is a
574 // workaround (is it?).
575 // I shall find a good way to avoid this once all constructors are converted
576 // to free methods
577 //
578 // In short, constructing a json by constructor calls to_json
579 // calling get calls from_json, for now, we cannot do this in custom
580 // serializers
581 nlohmann::from_json(j, value);
582 auto* bytes = static_cast<char*>(static_cast<void*>(&value)); // NOLINT(bugprone-casting-through-void)
583 std::memcpy(&t, bytes, sizeof(value));
584 }
585
586 template <
587 typename BasicJsonType, typename U = T,
588 typename std::enable_if <
589 !(std::is_pod<U>::value && std::is_class<U>::value), int >::type = 0 >
590 static void to_json(BasicJsonType& j, const T& t)
591 {
592 using nlohmann::to_json;
593 to_json(j, t);
594 }
595
596 template < typename BasicJsonType, typename U = T,
597 typename std::enable_if <
598 std::is_pod<U>::value && std::is_class<U>::value, int >::type = 0 >
599 static void to_json(BasicJsonType& j, const T& t) noexcept
600 {
601 const auto* bytes = static_cast< const unsigned char*>(static_cast<const void*>(&t)); // NOLINT(bugprone-casting-through-void)
602 std::uint64_t value = 0;
603 std::memcpy(&value, bytes, sizeof(value));
604 nlohmann::to_json(j, value);
605 }
606};

617struct non_pod
