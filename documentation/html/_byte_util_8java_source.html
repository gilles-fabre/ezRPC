11 // This produces a "hex dump" string representation of the given bytes.
12 // The output string is one or more lines, with 16 bytes printed on each line
13 // (except perhaps the last line). Each line shows the bytes in both hex format
14 // and ascii, e.g.
15 // 41 42 43 44 45 46 47 48 49 4A 4B 4C 4D 4E 4F 50 ABCDEFGHIJKLMNOP
16 public static String formatHexByteBuffer(byte buffer[], int offset, int size) {
17 if (size <= 0) {
18 return "\n";
19 }
20 int nLines = (size + (BYTES_PER_LINE-1)) / BYTES_PER_LINE;
21 byte[] buf = new byte[nLines * FORMATTED_CHARS_PER_LINE];
22 int i = 0; // index into input buffer
23 int j = 0; // index into output buf
24 for (int row = 0; row < nLines; row++) {
25 for (int col = 0; col < BYTES_PER_LINE; col++) {
26 byte h1, h2, ascii;
27 if (i < size) {
28 byte thisByte = buffer[offset + i];
29 h1 = hexDigits[(thisByte >> 4) & 0xF];
30 h2 = hexDigits[thisByte & 0xF];
31 ascii = makePrintable(thisByte);
32 } else {
33 h1 = ' ';
34 h2 = ' ';
35 ascii = ' ';
36 }
37 buf[j + (3*col) + 0] = h1;
38 buf[j + (3*col) + 1] = h2;
39 buf[j + (3*col) + 2] = ' ';
40 buf[j + OFFSET_TO_ASCII + col] = ascii;
41 i++;
42 }
43 buf[j + (OFFSET_TO_ASCII-2)] = ' ';
44 buf[j + (OFFSET_TO_ASCII-1)] = ' ';
45 buf[j + (FORMATTED_CHARS_PER_LINE-1)] = '\n';
46 j += FORMATTED_CHARS_PER_LINE;
47 }
48 return new String(buf);
