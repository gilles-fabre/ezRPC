87 private static void doItOnce(ModIdDefinitionDialog d, Component where, boolean isNewOp, String oldName) {
88 d.setVisible(true);
89 // We are blocked until the dialog is made invisible.
90 if (d.canceled) {
91 return;
92 }
93
94 String newName = d.nameTextField.getText();
95 int row = App.getMainFrame().getModIdByName(newName);
96 if (row >= 0 && (isNewOp || row != App.getMainFrame().getModIdByName(oldName))) {
97 d.errorString += '"' + newName + "\" " + Strings.getString("nameNotUnique");
98 d.hasErrors = true;

115 }
