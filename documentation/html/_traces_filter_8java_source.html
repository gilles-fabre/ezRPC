139 private boolean processIdMatches(int processId) {
140 matchPidFilter = null;
141 if (pidFilters.isEmpty())
142 return true;
143 Integer pId;
144 Iterator<Integer> iterator = pidFilters.iterator();
145 while (iterator.hasNext()) {
146 pId = (Integer) iterator.next();
147 if (pId.intValue() == processId) {
148 matchPidFilter = pId;
149 return true;
150 }
151 }
152 return false;
153 }

155 private boolean moduleIdMatches(int moduleId) {
156 matchModFilter = null;
157 if (modFilters.isEmpty())
158 return true;
159 ModuleFilter mId;
160 Iterator<ModuleFilter> iterator = modFilters.iterator();
161 while (iterator.hasNext()) {
162 mId = (ModuleFilter) iterator.next();
163 if (mId.moduleIdMatches(moduleId)) {
164 matchModFilter = mId;
165 matchColor = mId.color;

172 private boolean progMatches(String progName) {
173 matchProgFilter = null;
174 if (progFilters.isEmpty())
175 return true;
176 ProgFilter filter;
177 Iterator<ProgFilter> iterator = progFilters.iterator();
178 while (iterator.hasNext()) {
179 filter = (ProgFilter) iterator.next();
180 if (filter.nameMatches(progName)) {
181 matchProgFilter = filter;
182 matchColor = filter.color;
183 return true;
184 }
185 }
186 return false;

234 // Sort the module ids filters by their masks, so that more "specific" filters
235 // appear before less "specific" filters. We want to guarantee that if mask1
236 // "covers" mask2, i.e.
237 // (mask1 != mask2) && ((mask1 & mask2) == mask2)
238 // then mask1 precedes mask2 in the resulting list. It is sufficient to sort
239 // by the number of one bits in the mask, reversed.
240 public void sortModuleFiltersBySpecificity() {
241 Collections.sort(modFilters, new Comparator<ModuleFilter>() {
242 public int compare(ModuleFilter mf1, ModuleFilter mf2) {
243 return Integer.bitCount(mf2.modMask) - Integer.bitCount(mf1.modMask);
244 }});
245 }
