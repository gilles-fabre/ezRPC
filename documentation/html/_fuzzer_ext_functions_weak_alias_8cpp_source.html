19extern "C" {
20// Declare these symbols as weak to allow them to be optionally defined.
21#define EXT_FUNC(NAME, RETURN_TYPE, FUNC_SIG, WARN) \
22 RETURN_TYPE NAME##Def FUNC_SIG { \
23 Printf("ERROR: Function \"%s\" not defined.\n", #NAME); \
24 exit(1); \
25 } \
26 RETURN_TYPE NAME FUNC_SIG __attribute__((weak, alias(#NAME "Def")));

28#include "FuzzerExtFunctions.def"
29
30#undef EXT_FUNC
31}

33template <typename T>
34static T *GetFnPtr(T *Fun, T *FunDef, const char *FnName, bool WarnIfMissing) {
35 if (Fun == FunDef) {
36 if (WarnIfMissing)
37 Printf("WARNING: Failed to find function \"%s\".\n", FnName);
38 return nullptr;
39 }
40 return Fun;
41}
