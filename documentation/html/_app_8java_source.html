319 // Consume the queue of trace messages.
320 private void runInnards() {
321 while (running) {
322 // Block until there is at least one queued message.
323 // Acquire the rights to as many as possible up to a max of traceMessageQueueMaxBuffer.
324 traceMessageAvailable.acquireUninterruptibly();
325 int nMsg = Math.min(traceMessageQueueMaxBuffer, traceQueueSize());
326 if (nMsg > 1) {
327 traceMessageAvailable.acquireUninterruptibly(nMsg - 1);
328 }
329 // Remove the messages from the queue and buffer them to the content frames.
330 bufferQueuedMessages(nMsg);
331 // Have the EDT thread display the messages in the content frames.
332 try {
333 ContentFrameEnsemble.getInstance().displayBufferedMessages();
334 } catch (InterruptedException e) {
335 System.err.println("Consumer thread interrupted: " + e);
336 }
337 }
338 }
339 };

341 // Remove nMsg messages from the queue, and dispatch them to the content frames.
342 // This gets called on the consumer thread.
343 private void bufferQueuedMessages(int nMsg) {
344 // First remove the messages from the queue onto a separate list,
345 // so as to minimize the time that we hold the lock on the queue.
346 LinkedList<TraceQueueItem> bufferedMessageQueue = new LinkedList<TraceQueueItem>();
347 synchronized (traceMessageQueue) {
348 for (int i=0; i < nMsg; i++) {
349 bufferedMessageQueue.addLast(traceMessageQueue.removeFirst());
350 }
351 }
352 // Now process the messages we removed from the queue.
353 for (int i=0; i < nMsg; i++) {
354 TraceQueueItem item = bufferedMessageQueue.removeFirst();
355 if (item.isFirstMessage()) {
356 try {
357 dataThreadWithAFirstMessage = item.dataThread;
358 SwingUtilities.invokeAndWait(dataThreadFirstMessageHandler);
359 } catch (InvocationTargetException e) {
360 displayErrorAndDie(e);
361 } catch (InterruptedException e) {
362 System.err.println("Interrupted invoking dataThreadFirstMessageHandler: " + e);
363 }
364 }
365 ContentFrameEnsemble.getInstance().bufferIfMatches(item.message, item.dataThread);
366 }
367 }

369 private Runnable dataThreadFirstMessageHandler = new Runnable() {
370 // A data thread got its first message.
371 // - If we have a perThreadWindow policy then create a frame
372 // filtered with the pid and put the with "app name (pid)" as title.
373 // - Add a global program filter with the app name if it doesn't already exist.
374 // Reset the global time stamp base if this is the first message at all.
375 // - This method is called only once per data thread, on the EDT thread.
376 public void run() {
377 TraceMessage firstMessage = dataThreadWithAFirstMessage.getFirstMessage();
378 firstMessage.appName = transformProgName(firstMessage.appName);
379 // Note what's going on here. We have transformed the name within the first
380 // message within the data thread. But underneath us, in TracesDataThread.run,
381 // the transformed name will be copied from the first message into each new message.
382 System.out.println("dataThreadFirstMessageHandler: " + firstMessage.appName + " (" + firstMessage.processId + ")");
383 if (prefs.perThreadWindow) {
384 ContentFrame frm = ContentFrameEnsemble.getInstance().add();
385 frm.jSplitPane.setDividerLocation(0);
386 frm.setTitle(firstMessage.appName + " (" + firstMessage.processId + ")");
387 frm.dataThread = dataThreadWithAFirstMessage;
388 }
389 if (mainFrame != null) {
390 TracesFilter tf = mainFrame.getFilter();
391 ProgFilter pf = tf.getProgFilterByName(firstMessage.appName);
392 if (pf == null) {
393 // This program is not in the program filter palette, so add it there.
394 pf = new ProgFilter(firstMessage.appName, Color.BLACK);
395 pf.setPid(firstMessage.processId);
396 tf.addProgFilter(pf);
397 mainFrame.setFilterAndUpdateContentFrameFilters(tf);
398 } else {
399 // This program is already in the program filter palette, so update it with the pid.
400 mainFrame.updateProgFilterPid(pf.getName(), firstMessage.processId);
401 ContentFrameEnsemble.getInstance().updateProgFilterPid(pf.getName(), firstMessage.processId);
402 }
403 }
