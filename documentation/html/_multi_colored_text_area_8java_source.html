58 // Filter out lines that do NOT contain the regExp
59 private String filterOutLines(String s) {
60 String _regExp = regExpField.getText();
61 if (_regExp == null || _regExp.isEmpty())
62 return s;
63
64 if (!_regExp.equals(regExp) || pattern == null) {
65 try {
66 regExp = _regExp;
67 pattern = Pattern.compile(regExp);
68 } catch (PatternSyntaxException e) {
69 System.out.println("Misformatted regexp at index " + e.getIndex() + " : " + e.getDescription());
70 return s;
71 }
72 }

74 String result = "";
75 int start = 0, end = 0;
76 while ((end = s.indexOf('\n', start)) >= 0) {
77 String line = s.substring(start, end + 1);
78 start = end + 1;
79 Matcher m = pattern.matcher(line);
80 if (m.find())
81 result += line;

221 private int charIndexFromLineNumber(String text, int l) {
222 if (l <= 1) {
223 return 0;
224 }
225 int linesToSkip = l - 1;
226 int i = 0;
227 while (linesToSkip > 0) {
228 if (text.charAt(i) == '\n') {
