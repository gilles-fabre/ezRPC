304  // Consume the queue of trace messages.
305  private void runInnards() {
306  while (running) {
307  // Block until there is at least one queued message.
308  // Acquire the rights to as many as possible up to a max of traceMessageQueueMaxBuffer.
309  traceMessageAvailable.acquireUninterruptibly();
310  int nMsg = Math.min(traceMessageQueueMaxBuffer, traceQueueSize());
311  if (nMsg > 1) {
312  traceMessageAvailable.acquireUninterruptibly(nMsg - 1);
313  }
314  // Remove the messages from the queue and buffer them to the content frames.
315  bufferQueuedMessages(nMsg);
316  // Have the EDT thread display the messages in the content frames.
317  try {
318  ContentFrameEnsemble.getInstance().displayBufferedMessages();
319  } catch (InterruptedException e) {
320  System.err.println("Consumer thread interrupted: " + e);
321  }
322  }
323  }
324  };

326  // Remove nMsg messages from the queue, and dispatch them to the content frames.
327  // This gets called on the consumer thread.
328  private void bufferQueuedMessages(int nMsg) {
329  // First remove the messages from the queue onto a separate list,
330  // so as to minimize the time that we hold the lock on the queue.
331  LinkedList<TraceQueueItem> bufferedMessageQueue = new LinkedList<TraceQueueItem>();
332  synchronized (traceMessageQueue) {
333  for (int i=0; i < nMsg; i++) {
334  bufferedMessageQueue.addLast(traceMessageQueue.removeFirst());
335  }
336  }
337  // Now process the messages we removed from the queue.
338  for (int i=0; i < nMsg; i++) {
339  TraceQueueItem item = bufferedMessageQueue.removeFirst();
340  if (item.isFirstMessage()) {
341  try {
342  dataThreadWithAFirstMessage = item.dataThread;
343  SwingUtilities.invokeAndWait(dataThreadFirstMessageHandler);
344  } catch (InvocationTargetException e) {
345  displayErrorAndDie(e);
346  } catch (InterruptedException e) {
347  System.err.println("Interrupted invoking dataThreadFirstMessageHandler: " + e);
348  }
349  }
350  ContentFrameEnsemble.getInstance().bufferIfMatches(item.message, item.dataThread);
351  }
352  }

354  private Runnable dataThreadFirstMessageHandler = new Runnable() {
355  // A data thread got its first message.
356  // - If we have a perThreadWindow policy then create a frame
357  // filtered with the pid and put the with "app name (pid)" as title.
358  // - Add a global program filter with the app name if it doesn't already exist.
359  // Reset the global time stamp base if this is the first message at all.
360  // - This method is called only once per data thread, on the EDT thread.
361  public void run() {
362  TraceMessage firstMessage = dataThreadWithAFirstMessage.getFirstMessage();
363  firstMessage.appName = transformProgName(firstMessage.appName);
364  // Note what's going on here. We have transformed the name within the first
365  // message within the data thread. But underneath us, in TracesDataThread.run,
366  // the transformed name will be copied from the first message into each new message.
367  System.out.println("dataThreadFirstMessageHandler: " + firstMessage.appName + " (" + firstMessage.processId + ")");
368  if (prefs.perThreadWindow) {
369  ContentFrame frm = ContentFrameEnsemble.getInstance().add();
370  frm.jSplitPane.setDividerLocation(0);
371  frm.setTitle(firstMessage.appName + " (" + firstMessage.processId + ")");
372  frm.dataThread = dataThreadWithAFirstMessage;
373  }
374  if (mainFrame != null) {
375  TracesFilter tf = mainFrame.getFilter();
376  ProgFilter pf = tf.getProgFilterByName(firstMessage.appName);
377  if (pf == null) {
378  // This program is not in the program filter palette, so add it there.
379  pf = new ProgFilter(firstMessage.appName, Color.BLACK);
380  pf.setPid(firstMessage.processId);
381  tf.addProgFilter(pf);
382  mainFrame.setFilterAndUpdateContentFrameFilters(tf);
383  } else {
384  // This program is already in the program filter palette, so update it with the pid.
385  mainFrame.updateProgFilterPid(pf.getName(), firstMessage.processId);
386  ContentFrameEnsemble.getInstance().updateProgFilterPid(pf.getName(), firstMessage.processId);
387  }
388  }
389  }
